    <script>
        document.getElementById('predictionForm').addEventListener('submit', async function(event) {
            event.preventDefault();

            const form = event.target;
            const loadingIndicator = document.getElementById('loadingIndicator');
            const resultContainer = document.getElementById('resultContainer');
            const predictionResult = document.getElementById('predictionResult');

            // Show loading indicator and hide previous result
            loadingIndicator.classList.remove('hidden');
            resultContainer.classList.add('hidden');
            
            // Collect form data
            const formData = {
                age: parseInt(form.age.value),
                sex: form.sex.value,
                bmi: parseFloat(form.bmi.value),
                children: parseInt(form.children.value),
                smoker: form.smoker.value,
                region: form.region.value,
            };

            // This is where you will put the URL of your Vercel deployment
            // It will look something like: https://your-app-name.vercel.app/predict
            const apiUrl = '/api/predict';

            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(formData)
                });
                
                if (!response.ok) {
                    const errorData = await response.json();
                    throw new Error(errorData.error || `Server responded with status: ${response.status}`);
                }

                const data = await response.json();
                
                // Display the prediction result
                predictionResult.textContent = `$${data.premium}`;
                resultContainer.classList.remove('hidden');

            } catch (error) {
                console.error('Error:', error);
                predictionResult.textContent = `Error: ${error.message}`;
                resultContainer.classList.remove('hidden');
            } finally {
                loadingIndicator.classList.add('hidden');
            }
        });
    </script>
